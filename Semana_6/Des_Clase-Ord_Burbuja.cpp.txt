#include <iostream>

using namespace std;

void ordBurbujaModificado(int a[], int n)
{
	bool ordenado;
	for (int i = 0; i < n - 1; i++) //"i" solo nos sirve para el recorrido, no se usa como indice del arreglo 
	{
		cout << "\n" << "i = " << i << ", Valor = " << a[i] << endl; //Only For TEST
		ordenado = true;
		for (int j = 0; j < n - (i + 1); j++) //"j" se usa como indice del arreglo
		{
			if (a[j] > a[j + 1])
			{
				cout << "Intercambio: " << a[j] << " y " << a[j + 1] << endl; //Only For TEST
				int aux = a[j];
				a[j] = a[j + 1];
				a[j + 1] = aux;
				ordenado = false;
			}
		}
		if (ordenado) break; //Si en todas las pasadas del for "j", no cambia "ordenado(true)" => el arreglo ya est√° ordenado
	}
}


void main() {
	int numeros[] = { 4, 5, 2, 7, 8, 1, 0, 3, 6 };
	int N = 9;
	ordBurbujaModificado(numeros, N);

	int j = 0;
	while (j < N) {
		cout << numeros[j] << " ";
		j++;
	}
	cin.get();
}
